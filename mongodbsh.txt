#!/bin/bash

start() {  
/opt/module/mongodb/mongodb/bin/mongod  --config /opt/module/mongodb/mongod.conf 
}  
  
stop() {  
/opt/module/mongodb/mongodb/bin/mongod --config /opt/module/mongodb/mongod.conf --shutdown  
}  
case "$1" in  
  start)  
 start  
 ;;  
  
stop)  
 stop  
 ;;  
  
restart)  
 stop  
 start  
 ;;  
  *)  
 echo  
$"Usage: $0 {start|stop|restart}"  
 exit 1  
esac
